const fmtDate = (iso) => iso ? new Date(iso).toLocaleDateString("en-US", { month: "short", day: "numeric", year: "numeric" }) : "—";
const fmtTime = (iso) => iso ? new Date(iso).toLocaleTimeString("en-US", { hour: "2-digit", minute: "2-digit" }) : "—";
const fmtRelative = (iso) => {
  if (!iso) return "—";
  const diff = Date.now() - new Date(iso).getTime();
  if (diff < 60000) return "just now";
  if (diff < 3600000) return `${Math.floor(diff / 60000)}m ago`;
  if (diff < 86400000) return `${Math.floor(diff / 3600000)}h ago`;
  return `${Math.floor(diff / 86400000)}d ago`;
};

const useToast = () => {
  const [toasts, setToasts] = useState([]);

const Modal = ({ open, onClose, title, children, lg, xl }) => {
  useEffect(() => {
    if (open) document.body.style.overflow = "hidden";
    else document.body.style.overflow = "";
    return () => { document.body.style.overflow = ""; };
  }, [open]);

  useEffect(() => {
    if (!open) return;
    const handler = (e) => { if (e.key === "Escape") onClose(); };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, [open, onClose]);

  if (!open) return null;
  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className={`modal${lg ? " modal-lg" : ""}${xl ? " modal-xl" : ""}`} onClick={e => e.stopPropagation()}>
        <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 22, paddingBottom: 18, borderBottom: `1px solid ${T.border}` }}>
          <h2 className="f-orb" style={{ fontSize: 16, fontWeight: 700, color: T.t1, letterSpacing: "0.06em" }}>{title}</h2>
          <button onClick={onClose} className="btn-icon" style={{ fontSize: 14 }}>✕</button>
        </div>
        {children}
      </div>
    </div>
  );
};

            <feGaussianBlur stdDeviation="2" result="blur" />

        </defs>
        {/* Grid */}

          return (
            <line key={c.id} x1={x1} y1={y1} x2={x2} y2={y2}
              stroke={T.blue} strokeWidth={1} opacity={0.2} strokeDasharray="4 4" />
          );
        })}
        {/* Vehicles */}
        {vehicles.slice(0, 22).map(v => {
          const x = toX(v.lng), y = toY(v.lat);
          if (x < 8 || x > W - 8 || y < 8 || y > H - 8) return null;
          const color = statusColor[v.status] || T.t4;

          return (

                  <animate attributeName="opacity" values="0.6;0;0.6" dur="2.5s" repeatCount="indefinite" />

              )}

      </svg>
      {/* Legend */}
      <div style={{ position: "absolute", bottom: 10, left: 12, display: "flex", gap: 14 }}>

      </div>

  );

const FleetPage = ({ toast }) => {
  const [vehicles, setVehicles] = useState(_vehicles);
  const [search, setSearch] = useState("");
  const [statusFilter, setStatusFilter] = useState("");
  const [regionFilter, setRegionFilter] = useState("");
  const [showAdd, setShowAdd] = useState(false);
  const [selectedVehicle, setSelectedVehicle] = useState(null);
  const [page, setPage] = useState(1);
  const [sortKey, setSortKey] = useState("registration");
  const [sortDir, setSortDir] = useState(1);
  const PER_PAGE = 10;

  const filtered = useMemo(() => {
    const q = search.toLowerCase();
    return vehicles
      .filter(v => {
        const matchSearch = !q || v.registration.toLowerCase().includes(q) || v.type.toLowerCase().includes(q) || v.driver.toLowerCase().includes(q) || v.make.toLowerCase().includes(q) || v.region.toLowerCase().includes(q);
        const matchStatus = !statusFilter || v.status === statusFilter;
        const matchRegion = !regionFilter || v.region === regionFilter;
        return matchSearch && matchStatus && matchRegion;
      })
      .sort((a, b) => {
        const av = a[sortKey], bv = b[sortKey];
        if (typeof av === "number") return (av - bv) * sortDir;
        return String(av).localeCompare(String(bv)) * sortDir;
      });
  }, [vehicles, search, statusFilter, regionFilter, sortKey, sortDir]);

  const totalPages = Math.max(1, Math.ceil(filtered.length / PER_PAGE));
  const paginated = filtered.slice((page - 1) * PER_PAGE, page * PER_PAGE);

  useEffect(() => { setPage(1); }, [search, statusFilter, regionFilter]);

  const handleDelete = (id) => {
    if (!window.confirm("Remove this vehicle from the fleet?")) return;
    _vehicles = _vehicles.filter(v => v.id !== id);
    setVehicles([..._vehicles]);
    toast("Vehicle removed from fleet", "success");
  };

  const handleStatusChange = (id, newStatus) => {
    _vehicles = _vehicles.map(v => v.id === id ? { ...v, status: newStatus } : v);
    setVehicles([..._vehicles]);
    toast(`Status updated to ${newStatus}`, "success");
  };

  const handleAdd = (data) => {
    const v = {
      id: uuid(), ...data,
      status: "idle", lastPing: new Date().toISOString(),
      lat: -1.286389 + (Math.random() - 0.5) * 6,
      lng: 30 + Math.random() * 10,
      fuelLevel: 80, mileage: 0, speed: 0, heading: 0,
      engineHours: 0, nextService: new Date(Date.now() + 30 * 86400000).toISOString(),
    };
    _vehicles = [v, ..._vehicles];
    setVehicles([..._vehicles]);
    setShowAdd(false);
    toast("Vehicle added to fleet", "success");
  };

  const toggleSort = (key) => {
    if (sortKey === key) setSortDir(d => -d);
    else { setSortKey(key); setSortDir(1); }
  };

  const SortTh = ({ label, k }) => (
    <th style={{ cursor: "pointer", userSelect: "none" }} onClick={() => toggleSort(k)}>
      {label} {sortKey === k ? (sortDir > 0 ? "↑" : "↓") : ""}
    </th>
  );

  const statusCounts = STATUSES_V.reduce((acc, s) => ({ ...acc, [s]: vehicles.filter(v => v.status === s).length }), {});
  const statusColors = { active: T.green, idle: T.t3, maintenance: T.gold, deployed: T.blue };

  return (
    <div style={{ paddingBottom: 36 }}>
      <div className="page-header">
        <h1 className="page-title">Fleet Management</h1>
        <p className="page-sub">{filtered.length} vehicles · {vehicles.filter(v => v.status === "active").length} active</p>
      </div>

      {/* Status pills */}
      <div className="grid-4" style={{ marginBottom: 20 }}>
        {STATUSES_V.map(s => (
          <div key={s} className="stat-card" onClick={() => setStatusFilter(statusFilter === s ? "" : s)}
            style={{ "--accent": statusColors[s], "--accent-border": `${statusColors[s]}30`, padding: 16, cursor: "pointer",
              borderColor: statusFilter === s ? statusColors[s] : T.border }}>
            <div className="stat-card-accent" style={{ background: statusColors[s] }} />
            <p style={{ fontSize: 11, color: T.t3, textTransform: "uppercase", letterSpacing: "0.08em", marginBottom: 6, fontWeight: 600 }}>{s}</p>
            <p className="stat-value" style={{ fontSize: 28, color: statusFilter === s ? statusColors[s] : T.t1 }}>{statusCounts[s]}</p>
          </div>
        ))}
      </div>

      {/* Filters */}
      <div className="card" style={{ marginBottom: 16, display: "flex", gap: 12, flexWrap: "wrap", padding: 14, alignItems: "center" }}>
        <div className="search-box" style={{ flex: "1 1 260px", minWidth: 200 }}>
          <span className="search-icon">⌕</span>
          <input className="input" placeholder="Search vehicles, drivers, plates…" value={search} onChange={e => setSearch(e.target.value)} />
        </div>
        <select className="input" value={statusFilter} onChange={e => setStatusFilter(e.target.value)} style={{ minWidth: 140 }}>
          <option value="">All Status</option>
          {STATUSES_V.map(s => <option key={s} value={s}>{s.charAt(0).toUpperCase() + s.slice(1)}</option>)}
        </select>
        <select className="input" value={regionFilter} onChange={e => setRegionFilter(e.target.value)} style={{ minWidth: 140 }}>
          <option value="">All Regions</option>
          {REGIONS.map(r => <option key={r} value={r}>{r}</option>)}
        </select>
        <button className="btn btn-primary" onClick={() => setShowAdd(true)}>+ Add Vehicle</button>
      </div>

      {/* Table */}
      <div className="card" style={{ padding: 0, overflow: "hidden" }}>
        <div className="table-wrap">
          <table>
            <thead>
              <tr>
                <SortTh label="Registration" k="registration" />
                <SortTh label="Type / Make" k="type" />
                <th>Driver</th>
                <SortTh label="Region" k="region" />
                <th>Status</th>
                <SortTh label="Fuel" k="fuelLevel" />
                <SortTh label="Speed" k="speed" />
                <th>Last Ping</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {paginated.length === 0 ? (
                <tr><td colSpan={9} style={{ textAlign: "center", padding: 48, color: T.t3 }}>No vehicles match your filters</td></tr>
              ) : paginated.map(v => (
                <tr key={v.id} onClick={() => setSelectedVehicle(v)} style={{ cursor: "pointer" }}>
                  <td><span className="f-mono" style={{ color: T.gold, fontWeight: 700 }}>{v.registration}</span></td>
                  <td>
                    <p style={{ fontWeight: 600, color: T.t1 }}>{v.type}</p>
                    <p style={{ fontSize: 11.5, color: T.t3, marginTop: 1 }}>{v.make}</p>
                  </td>
                  <td style={{ color: T.t2 }}>{v.driver}</td>
                  <td style={{ color: T.t2 }}>{v.region}</td>
                  <td><StatusBadge status={v.status} /></td>
                  <td><FuelIndicator level={v.fuelLevel} /></td>
                  <td>
                    <span className="f-mono" style={{ color: v.speed > 100 ? T.red : T.t2, fontSize: 13 }}>
                      {v.speed} km/h
                    </span>
                  </td>
                  <td style={{ color: T.t3, fontSize: 12.5 }}>{fmtRelative(v.lastPing)}</td>
                  <td onClick={e => e.stopPropagation()}>
                    <div style={{ display: "flex", gap: 5 }}>
                      <button className="btn-icon" title="View Details" onClick={() => setSelectedVehicle(v)}>👁</button>
                      <select className="input" value={v.status}
                        onChange={e => handleStatusChange(v.id, e.target.value)}
                        style={{ padding: "4px 8px", fontSize: 12, minWidth: 0, width: "auto", height: 34 }}
                        title="Change Status" onClick={e => e.stopPropagation()}>
                        {STATUSES_V.map(s => <option key={s} value={s}>{s}</option>)}
                      </select>
                      <button className="btn-icon" style={{ color: T.red, borderColor: "rgba(242,82,82,0.25)" }} title="Remove" onClick={() => handleDelete(v.id)}>✕</button>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
        {totalPages > 1 && (
          <div style={{ display: "flex", justifyContent: "center", alignItems: "center", gap: 14, padding: "14px 20px", borderTop: `1px solid ${T.border}` }}>
            <button className="btn btn-secondary btn-sm" disabled={page <= 1} onClick={() => setPage(p => p - 1)}>← Prev</button>
            <span style={{ fontSize: 12.5, color: T.t3 }}>
              <span className="f-mono">{page}</span> / <span className="f-mono">{totalPages}</span>
              <span style={{ marginLeft: 10, color: T.t4 }}>({filtered.length} total)</span>
            </span>
            <button className="btn btn-secondary btn-sm" disabled={page >= totalPages} onClick={() => setPage(p => p + 1)}>Next →</button>
          </div>
        )}
      </div>

      <Modal open={showAdd} onClose={() => setShowAdd(false)} title="Add Vehicle to Fleet">
        <AddVehicleForm onSubmit={handleAdd} onCancel={() => setShowAdd(false)} />
      </Modal>

      <Modal open={!!selectedVehicle} onClose={() => setSelectedVehicle(null)} title={selectedVehicle?.registration || ""} lg>
        {selectedVehicle && <VehicleDetail vehicle={selectedVehicle} />}
      </Modal>
    </div>
  );
};

const AlertsPage = ({ toast }) => {
  const [alerts, setAlerts] = useState(_alerts);
  const [severityFilter, setSeverityFilter] = useState("");
  const [typeFilter, setTypeFilter] = useState("");
  const [showResolved, setShowResolved] = useState(false);
  const [search, setSearch] = useState("");

  const filtered = useMemo(() => {
    const q = search.toLowerCase();
    return alerts.filter(a => {
      if (!showResolved && a.resolved) return false;
      if (showResolved && !a.resolved) return false;
      if (severityFilter && a.severity !== severityFilter) return false;
      if (typeFilter && a.type !== typeFilter) return false;
      if (q && !a.vehicleReg.toLowerCase().includes(q) && !a.message.toLowerCase().includes(q) && !a.region.toLowerCase().includes(q)) return false;
      return true;
    });
  }, [alerts, severityFilter, typeFilter, showResolved, search]);

  const handleAcknowledge = (id) => {
    _alerts = _alerts.map(a => a.id === id ? { ...a, resolved: true, acknowledgedBy: "Operator" } : a);
    setAlerts([..._alerts]);
    toast("Alert acknowledged", "success");
  };

  const handleAcknowledgeAll = () => {
    _alerts = _alerts.map(a => ({ ...a, resolved: true, acknowledgedBy: "Operator" }));
    setAlerts([..._alerts]);
    toast("All alerts cleared", "success");
  };

  const sevColor = { critical: T.red, high: T.red, medium: T.gold, low: T.green };
  const unresolved = alerts.filter(a => !a.resolved).length;

  const sevCounts = ALERT_SEV.reduce((acc, s) => ({
    ...acc, [s]: alerts.filter(a => a.severity === s && !a.resolved).length
  }), {});

  return (
    <div style={{ paddingBottom: 36 }}>
      <div className="page-header">
        <h1 className="page-title">Alert Center</h1>
        <p className="page-sub">{unresolved} active alerts requiring attention</p>
      </div>

      {/* Severity stats */}
      <div className="grid-4" style={{ marginBottom: 20 }}>
        {ALERT_SEV.map(s => (
          <div key={s} className="stat-card" style={{ "--accent": sevColor[s], padding: 16, cursor: "pointer",
            borderColor: severityFilter === s ? sevColor[s] : T.border }}
            onClick={() => setSeverityFilter(severityFilter === s ? "" : s)}>
            <div className="stat-card-accent" style={{ background: sevColor[s] }} />
            <p style={{ fontSize: 11, color: T.t3, textTransform: "uppercase", letterSpacing: "0.08em", marginBottom: 6, fontWeight: 600 }}>{s}</p>
            <p className="stat-value" style={{ fontSize: 28, color: severityFilter === s ? sevColor[s] : T.t1 }}>{sevCounts[s]}</p>
          </div>
        ))}
      </div>

      {/* Filters */}
      <div className="card" style={{ marginBottom: 16, display: "flex", gap: 12, flexWrap: "wrap", padding: 14, alignItems: "center" }}>
        <div className="search-box" style={{ flex: "1 1 200px" }}>
          <span className="search-icon">⌕</span>
          <input className="input" placeholder="Search alerts…" value={search} onChange={e => setSearch(e.target.value)} />
        </div>
        <select className="input" value={severityFilter} onChange={e => setSeverityFilter(e.target.value)} style={{ minWidth: 140 }}>
          <option value="">All Severity</option>
          {ALERT_SEV.map(s => <option key={s} value={s}>{s.charAt(0).toUpperCase() + s.slice(1)}</option>)}
        </select>
        <select className="input" value={typeFilter} onChange={e => setTypeFilter(e.target.value)} style={{ minWidth: 140 }}>
          <option value="">All Types</option>
          {ALERT_TYPES.map(t => <option key={t} value={t}>{t.charAt(0).toUpperCase() + t.slice(1)}</option>)}
        </select>
        <div className="tabs" style={{ minWidth: 220 }}>
          <button className={`tab${!showResolved ? " active" : ""}`} onClick={() => setShowResolved(false)}>Active ({unresolved})</button>
          <button className={`tab${showResolved ? " active" : ""}`} onClick={() => setShowResolved(true)}>Resolved ({alerts.length - unresolved})</button>

      {filtered.length === 0 ? (

      ) : (
        <div style={{ display: "flex", flexDirection: "column", gap: 9 }}>
          {filtered.map(a => (
            <div key={a.id} className="alert-item" style={{ opacity: a.resolved ? 0.55 : 1 }}>
              <div style={{ display: "flex", alignItems: "flex-start", paddingTop: 3 }}>
                <span style={{ width: 10, height: 10, borderRadius: "50%", background: sevColor[a.severity] || T.gold, flexShrink: 0 }} />

const AuditPage = () => {
  const [logs] = useState(_auditLog);
  const [search, setSearch] = useState("");

  const filtered = useMemo(() => {
    const q = search.toLowerCase();
    return logs.filter(l => !q || l.action.toLowerCase().includes(q) || l.user.toLowerCase().includes(q) || l.target.toLowerCase().includes(q));
  }, [logs, search]);

  const actionIcon = (action) => {
    if (action.includes("added")) return { icon: "➕", color: T.green, bg: T.greenDim };
    if (action.includes("removed")) return { icon: "✕", color: T.red, bg: T.redDim };
    if (action.includes("created")) return { icon: "🔖", color: T.blue, bg: T.blueDim };
    if (action.includes("changed") || action.includes("updated")) return { icon: "↻", color: T.gold, bg: T.goldGlow };
    if (action.includes("acknowledged")) return { icon: "✓", color: T.green, bg: T.greenDim };
    return { icon: "📋", color: T.t3, bg: "rgba(75,96,128,0.15)" };
  };

  return (
    <div style={{ paddingBottom: 36 }}>
      <div className="page-header">
        <h1 className="page-title">Audit Log</h1>
        <p className="page-sub">Complete activity trail for all fleet operations</p>
      </div>

      <div style={{ marginBottom: 16 }}>
        <div className="search-box">
          <span className="search-icon">⌕</span>
          <input className="input" placeholder="Search logs by action, user, or target…" value={search} onChange={e => setSearch(e.target.value)} />
        </div>
      </div>

      <div className="card" style={{ padding: 0, overflow: "hidden" }}>
        <div className="table-wrap">
          <table>
            <thead>
              <tr>
                <th>Action</th>
                <th>User</th>
                <th>Target</th>
                <th>IP</th>
                <th>Time</th>
              </tr>
            </thead>
            <tbody>
              {filtered.map(log => {
                const meta = actionIcon(log.action);
                return (
                  <tr key={log.id}>
                    <td>
                      <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
                        <span style={{ width: 28, height: 28, borderRadius: 7, background: meta.bg, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 13, color: meta.color, flexShrink: 0 }}>{meta.icon}</span>
                        <span style={{ color: T.t1, fontWeight: 500 }}>{log.action}</span>
                      </div>
                    </td>
                    <td style={{ color: T.t2 }}>{log.user}</td>
                    <td><span className="f-mono" style={{ fontSize: 12, color: T.gold }}>{log.target}</span></td>
                    <td><span className="f-mono" style={{ fontSize: 12, color: T.t4 }}>{log.ip}</span></td>
                    <td style={{ color: T.t3, fontSize: 12.5 }}>{fmtRelative(log.ts)}</td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};
